rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Config collection - public read
    match /config/{document} {
      allow read: if true;
      allow write: if false;
    }
    
    // Properties - public read (for homepage), authenticated write
    match /properties/{propertyId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Packages - public read, authenticated write
    match /packages/{packageId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Users - authenticated read/write
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    
    // Users favorites - authenticated read/write
    match /users/{userId}/favorites/{favoriteId} {
      allow read, write: if request.auth != null;
    }
    
    // Wallet - authenticated read/write
    match /wallet/{userId} {
      allow read, write: if request.auth != null;
    }
    
    // Wallet transactions - authenticated read/write
    match /wallet/{userId}/transactions/{transactionId} {
      allow read, write: if request.auth != null;
    }
    
    // Saved searches - authenticated read/write
    match /saved_searches/{searchId} {
      allow read, write: if request.auth != null;
    }
    
    // Conversations - authenticated read/write
    match /conversations/{conversationId} {
      allow read, write: if request.auth != null;
    }
    
    // Messages - authenticated read/write
    match /conversations/{conversationId}/messages/{messageId} {
      allow read, write: if request.auth != null;
    }
    
    // Analytics - authenticated read/write
    match /analytics/{document} {
      allow read, write: if request.auth != null || true;
    }
    
    // Analytics clicks - authenticated read/write
    match /analytics/{document}/clicks/{clickId} {
      allow read, write: if request.auth != null || true;
    }
    
    // User wallet cache - authenticated read/write
    match /user_wallet_cache/{userId} {
      allow read, write: if request.auth != null;
    }
    
    // User wallet cache transactions - authenticated read/write
    match /user_wallet_cache/{userId}/cached_transactions/{transactionId} {
      allow read, write: if request.auth != null;
    }
  }
}
