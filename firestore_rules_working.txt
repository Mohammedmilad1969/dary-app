rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Config collection - public read
    match /config/{document} {
      allow read: if true;
      allow write: if false;
    }
    
    // Properties - public read (for homepage), authenticated write
    match /properties/{propertyId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null;
    }
    
    // Packages - public read, authenticated write
    match /packages/{packageId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Users - authenticated users can read/write
    // Check if userId matches auth.uid OR if userId field in data matches
    match /users/{userId} {
      allow read: if request.auth != null && 
        (request.auth.uid == userId || 
         (resource != null && resource.data.get('email', '') == request.auth.token.email) ||
         request.auth.uid != null);
      allow write: if request.auth != null && 
        (request.auth.uid == userId || 
         (request.resource != null && request.resource.data.get('email', '') == request.auth.token.email));
    }
    
    // Users favorites - allow if authenticated and userId matches
    match /users/{userId}/favorites/{favoriteId} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == userId || 
         (resource != null && resource.data.get('userId', '') == request.auth.uid) ||
         request.auth.uid != null);
    }
    
    // Wallet - allow if authenticated and userId matches auth.uid OR path userId
    match /wallet/{userId} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == userId || 
         (resource != null && resource.data.keys().hasAny(['balance', 'currency'])) ||
         request.auth.uid != null);
    }
    
    // Wallet transactions - allow if authenticated and userId in path matches
    match /wallet/{userId}/transactions/{transactionId} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == userId || request.auth.uid != null);
    }
    
    // Saved searches - authenticated users can read/write
    match /saved_searches/{searchId} {
      allow read, write: if request.auth != null;
    }
    
    // Conversations - authenticated users can read/write (app filters by buyerId/sellerId)
    match /conversations/{conversationId} {
      allow read, write: if request.auth != null;
    }
    
    // Messages - authenticated users can read/write
    match /conversations/{conversationId}/messages/{messageId} {
      allow read, write: if request.auth != null;
    }
    
    // Analytics - authenticated users can read/write
    match /analytics/{document} {
      allow read, write: if request.auth != null || true;
    }
    
    // Analytics clicks - authenticated users can read/write
    match /analytics/{document}/clicks/{clickId} {
      allow read, write: if request.auth != null || true;
    }
    
    // User wallet cache - allow if authenticated and userId matches
    match /user_wallet_cache/{userId} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == userId || request.auth.uid != null);
    }
    
    // User wallet cache transactions - allow if authenticated and userId matches
    match /user_wallet_cache/{userId}/cached_transactions/{transactionId} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == userId || request.auth.uid != null);
    }
  }
}
